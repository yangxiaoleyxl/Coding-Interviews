10. 乘积最大子数组 
```python 
class Solution:
    def maxProduct(self, nums: List[int]) -> int: 
        if not nums: return  
        # 思路同, “最大子数组和”, 但是要新增维度   
        res = nums[0]
        pre_max = nums[0] # 记录最大值
        pre_min = nums[0] # 记录最小值
        for num in nums[1:]: 
            # 
            cur_max = max(pre_max*num, pre_min*num, num) 
            cur_min = min(pre_max*num, pre_min*num, num) 
            res = max(res, cur_max) 
            pre_max = cur_max 
            pre_min = cur_min 
        return res 
```
